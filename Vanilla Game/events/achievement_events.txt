# Viking Raider
unit_event = {
	id = 700
	hide_window = yes
	
	ai = no
	
	is_triggered_only = yes
	
	trigger = {
		FROM = {
			culture_group = north_germanic
			OR = {
				religion = norse_pagan
				religion = norse_pagan_reformed
			}
		}
		loot = 1000
	}
	
	immediate = {
		FROM = {
			if = {
				limit = {
					NOT = { has_character_flag = achievement_viking_raider }
				}
				set_character_flag = achievement_viking_raider
			}
		}
	}
	
	option = {
		name = OK
	}
}
